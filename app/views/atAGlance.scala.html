Concurrent Support
Use eventually with implicit PatienceConfig 
import org.scalatest.concurrent.Eventually._
eventually { ... }
Define custom implicit PatienceConfig 
import org.scalatest.concurrent.PatienceConfig
import org.scalatest.time._
implicit val patienceConfig =
PatienceConfig(timeout = scaled(Span(2, Seconds)), interval = scaled(Span(5, Millis)))
Use eventually with explicit timeout 
import org.scalatest.concurrent.Eventually._
import org.scalatest.time._
eventually (timeout(Span(5, Seconds))) { ... }
Use eventually with explicit timeout and interval 
import org.scalatest.concurrent.Eventually._
import org.scalatest.time._
eventually (timeout(Span(5, Seconds)), interval(Span(5, Millis))) { ... }
Use SpanSugar to specify time 
import org.scalatest.concurrent.Eventually._

Inspectors
Use forAll to check that every element passes the assertion block 
import org.scalatest._
import Matchers._
import Inspectors._
forAll(List(1, 2, 3)) { _ should be > 0 }
Use forAtLeast to check that at least the specified number of elements pass the assertion block 
import org.scalatest._
import Matchers._
import Inspectors._
forAtLeast(1, List(1, 2, 3)) { _ should be > 2 }
Use forAtMost to check that at most the specified number of elements pass the assertion block 
import org.scalatest._
import Matchers._
import Inspectors._
forAtMost(2, List(1, 2, 3)) { _ should be > 1 }
Use forBetween to check that the specified minimum and maximum number of elements (inclusive) pass the assertion block 
import org.scalatest._
import Matchers._
import Inspectors._
forBetween(2, 4, List(1, 2, 3, 4, 5)) { _ should be > 2 }
Use forEvery to check that every element passes the assertion block, it'll list all failing elements if it fails (whereas forAll just reports the first failing element) 
import org.scalatest._
import Matchers._
import Inspectors._
forEvery(List(1, 2, 3)) { _ should be > 0 }
Use forExactly to check that the exact specified number of elements pass the assertion block 
import org.scalatest._
import Matchers._
import Inspectors._
forExactly(2, List(1, 2, 3)) { _ should be > 1 }
